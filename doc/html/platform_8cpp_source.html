63  for (auto const &filename : compFilenames) {
64  dict content = importer.importAsDict(dir + filename);
65 
66  string compType = getContentType(content, dir);
67 
68  if (factoryMap.count(compType) == 0) {
69  // If no constructor was defined for this component type...
70  throw std::runtime_error("Error: unknown compType \"" + compType + "\"\n");
71  }
72 
73  spdlog::info("Loading " + compType);
74 
75  // Create new component of base class type
76  shared_ptr<Component> newComp = factoryMap[compType]->makeFromFileContent(content);
77 
78  if (content.count(constants::SOURCE) > 0) { // Bindable type
79  // Cast the new comp to the child type and save it separately
80  bindableComps.push_back(
81  std::dynamic_pointer_cast<IBindableComponent>(newComp)

83  }
84  if (content.count(constants::LABEL)) { // Readable type
85  readableComps.push_back(
86  std::dynamic_pointer_cast<IReadableComponent>(newComp)

88  }
89  components.push_back(newComp);
90  addDependencies(components.back(), content);
91  }

145 std::string HPS::getContentType(dict &content, const string rootPath) {
146  if (content.count(constants::TYPE) > 0) {
147  return content[constants::TYPE];
148  }
149  for (auto const &key : content) {
150  // If one key is a filename, we consider it as a Platform
151  std::ifstream f((rootPath + key.first).c_str());
152  if (f.good()) {
